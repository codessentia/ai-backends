Ready-to-use AI Services

Access a suite of specialized AI models through a unified API, with just a few lines of code

example.js 
// npm install aibackends
import { AIBackends } from "aibackends";

const aibackends = AIBackends({
    apiKey: "sk39wo393.....32ncsmw9339RNj3" // Retrieve API key from dashboard
});

const response = await aibackends.web.search({
    "query": "What are the best food places to visit in San Francisco?"
});
# pip install aistack
from aibackends import AIBackends

aibackends = AIBackends(api_key="sk39wo393.....32ncsmw9339RNj3")

response = aibackends.web.search(
    query="What are the best food places to visit in San Francisco?"
)
